Install dependencies
Required Python libraries for the full segmentation pipeline.

2
Load and prepare transaction data
Load your CSV of transactions and build the RFM feature matrix.
import pandas as pd import numpy as np from datetime import datetime # Load transactions df = pd.read_csv('transactions.csv', parse_dates=['order_date']) TODAY = pd.Timestamp('2025-06-01') # Build RFM table rfm = df.groupby('customer_id').agg( recency=('order_date', lambda x: (TODAY - x.max()).days), frequency=('order_id', 'count'), monetary=('revenue', 'sum') ).reset_index() print(rfm.describe())
3
Feature scaling & optimal K selection
Standardize features then use the elbow method to find optimal cluster count.
from sklearn.preprocessing import StandardScaler from sklearn.cluster import KMeans from kneed import KneeLocator scaler = StandardScaler() X = scaler.fit_transform(rfm[['recency', 'frequency', 'monetary']]) # Elbow method inertias = [] K_range = range(2, 11) for k in K_range: km = KMeans(n_clusters=k, random_state=42, n_init=10) km.fit(X) inertias.append(km.inertia_) kl = KneeLocator(K_range, inertias, curve='convex', direction='decreasing') optimal_k = kl.knee print(f"Optimal K: {optimal_k}")
4
Fit K-Means and label segments
Cluster customers and assign business-friendly segment names based on RFM scores.
# Fit final model kmeans = KMeans(n_clusters=5, random_state=42, n_init=10) rfm['cluster'] = kmeans.fit_predict(X) # Score each cluster to assign labels def label_cluster(row): r, f, m = row['recency'], row['frequency'], row['monetary'] if r < 30 and f > 6 and m > 3000: return 'Champions' elif r < 60 and f > 3: return 'Loyal Customers' elif r < 90 and f >= 1: return 'Potential Loyalists' elif r > 60 and f > 2: return 'At Risk' else: return 'New Customers' rfm['segment'] = rfm.apply(label_cluster, axis=1) print(rfm['segment'].value_counts())

Visualize results with Plotly
Create an interactive 3D scatter plot of your RFM clusters.
import plotly.express as px fig = px.scatter_3d( rfm, x='recency', y='frequency', z='monetary', color='segment', opacity=0.7, size_max=8, color_discrete_map={ 'Champions': '#00c896', 'Loyal Customers': '#4a9eff', 'Potential Loyalists': '#f5a623', 'At Risk': '#ff5b5b', 'New Customers': '#9d7bff' }, title='Customer Segments — 3D RFM Space', template='plotly_dark' ) fig.show()
